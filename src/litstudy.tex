% Making sure a term is defined before use, and defined only once:
\makeatletter % Allow usage of '@' in macro names
% For defining a term: if it's not defined before, typeset it correctly, otherwise error.
\renewcommand*{\newterm}[1]{%
  % If the term specialtermdef@TERM isn't defined
  \@ifundefined{specialtermdef@\detokenize\expandafter{#1}}
  % Define it
  {\@namedef{specialtermdef@\detokenize\expandafter{#1}}{}%
  % Typeset the term in italics
  \emph{#1}}
  % Otherwise, error: double definition
  {\PackageError{termtrack}{Term '#1' already defined}{Double definition}}}

% For using a defined term: if it hasn't been defined, error.
\renewcommand*{\term}[1]{%
  % If the term specialtermdef@TERM isn't defined
  \@ifundefined{specialtermdef@\detokenize{#1}}
  % Error: using a term before defining it
  {\PackageError{termtrack}{Term '#1' not defined}{Use \protect\defterm\space macro}}
  % Otherwise, typeset it
  {#1}}
\makeatother % Make a '@' a regular letter again
